230  """

259 def CreateOperator(
260  operator_type,
261  inputs,
262  outputs,
263  name='',
264  control_input=None,
265  device_option=None,
266  arg=None,
267  engine=None,
268  **kwargs
269 ):
270  """A function wrapper that allows one to create operators based on the
271  operator type. The type should be a string corresponding to an operator
272  registered with Caffe2.
273  """
274  operator = caffe2_pb2.OperatorDef()
275  operator.type = operator_type
276  operator.name = name
277  # Add rectified inputs and outputs
278  inputs = _RectifyInputOutput(inputs)
279  outputs = _RectifyInputOutput(outputs)
280  operator.input.extend([str(i) for i in inputs])
281  operator.output.extend([str(o) for o in outputs])
282  if control_input:
283  control_input = _RectifyInputOutput(control_input)
284  operator.control_input.extend([str(i) for i in control_input])
285  # Set device option:
286  # (1) If device_option is explicitly set, use device_option.
287  # (2) If not, but scope.CurrentDeviceScope() is set,
288  # then we use scope.CurrentDeviceScope().
289  # (3) Otherwise, do not set device option.
290  if device_option is not None:
291  operator.device_option.CopyFrom(device_option)
292  elif scope.CurrentDeviceScope() is not None:
293  operator.device_option.CopyFrom(scope.CurrentDeviceScope())
294  if engine is not None:
295  operator.engine = engine
296  # random seed is defined in the device option, so we need to do special
297  # care.
298  if 'random_seed' in kwargs:
299  operator.device_option.random_seed = kwargs['random_seed']
300  del kwargs['random_seed']
301  # Add given arguments that do not need parsing
302  if arg is not None:
303  operator.arg.extend(arg)
304  # Add all other arguments
305  for key, value in kwargs.items():
306  operator.arg.add().CopyFrom(utils.MakeArgument(key, value))

309  workspace.RunOperatorImmediate(operator)
310  return operator

377 class IR(object):
378  """A simple IR class to keep track of all intermediate representations used
379  in the gradient computation.
380  """
381 
382  def __init__(self, operators):
383  # The IR class holds multiple metadata from the forward pass:
384  # a) ssa: a list of [op, in_versions, out_versions] recording the
385  # input and the output version of each operator, similar
386  # to a normal SSA form.
387  # b) input_count: a dictionary specifying for each blob and
388  # each of its version, how many times it is used as input for another
389  # op.
390  # c) frontier: maintaining the current versions of the blobs
391  # we are having in the workspace, after the execution of all the ops
392  # added to the IR so far. This is useful because if a gradient is
393  # trying to access an earlier version of a blob, we can sanity check
394  # that it is no longer there, and thus throw an error.
395  # d) gradient_frontier: maps the names of blobs to its version that the
396  # gradient corresponds to.
397  # e) gradient_generators: for each blob and each of its version, maps to
398  # a list of operators that generates its gradient together with the
399  # gradient name.
400  self.ssa = []
401  self.input_usages = defaultdict(lambda: defaultdict(list))
402  self.frontier = defaultdict(int)
403  self.gradient_frontier = {}
404  self.gradient_generators = defaultdict(lambda: defaultdict(list))
405 
406  for op in operators:
407  self.Play(op)

409  def Play(self, op):
410  """"Adds an op to the current IR, and update the internal states to
411  reflect the blobs and versions after the execution of the op.
412  """
413  # For input, they are the current version in the dict.
414  in_versions = {}
415  for s in op.input:
416  in_versions[s] = self.frontier[s]
417  self.input_usages[s][self.frontier[s]].append(len(self.ssa))
418  # For output, they are the current version plus one. If this is a
419  # newly created blob, its version starts with zero.
420  out_versions = {}
421  for s in op.output:
422  if s in self.frontier:
423  self.frontier[s] += 1
424  out_versions[s] = self.frontier[s]
425  # Add to SSA for bookkeeping.
426  self.ssa.append(OpSSA(op, in_versions, out_versions))

428  def CheckGradientOperatorInput(
429  self, grad_op_input, g_output, fwd_op_idx, locally_generated_blobs):
430  """Checks if the gradient operators can be correctly carried out."""
431  forward_op, in_versions, out_versions = self.ssa[fwd_op_idx]
432  original_index = GetIndexFromGradientList(g_output, grad_op_input)
433  # If it is a dense or sparse gradient name, it should match the
434  # version of the corresponding output.
435  if original_index is not None:

439  raise RuntimeError(
440  'Gradient name "%s" is expected to correspond '
441  'to version %d of "%s", but currently we have '
442  'version %d.' % (
443  grad_op_input, out_versions[original_name],
444  original_name,
445  self.gradient_frontier[original_name]))
446  # If it is an output name, the current version should match the
447  # version when the operator was run.
448  elif grad_op_input in out_versions:
449  if self.frontier[grad_op_input] != out_versions[grad_op_input]:
450  raise RuntimeError(

453  grad_op_input, out_versions[grad_op_input],

455  )
456  )
457  # If it is an input name, the current version should match the
458  # version when the operator was run.
459  elif grad_op_input in in_versions:
460  if (self.frontier[grad_op_input] != in_versions[grad_op_input]):
461  raise RuntimeError(

501  def BuildGradientGenerators( # NOQA
502  self, fwd_op_idx, gradient_ops, g_output, g_input):
503  """Updates gradient_generators and gradient_frontier"""
504  forward_op, in_versions, out_versions = self.ssa[fwd_op_idx]
505  locally_generated_blobs = []
506  sparse_generators = defaultdict(lambda: defaultdict(list))
507 
508  for grad_op in gradient_ops:
509  # (1) check that inputs are valid
510  for s in grad_op.input:
511  self.CheckGradientOperatorInput(
512  s, g_output, fwd_op_idx, locally_generated_blobs)
513 
514  # (2) add outputs to the locally generated blobs
515  # If an output corresponds to the gradient of an input, we also
516  # record it to gradient_generators
517  locally_generated_blobs.extend([str(s) for s in grad_op.output])
518  for i, output in enumerate(grad_op.output):
519  input_index = GetIndexFromGradientList(g_input, output)
520  if input_index is not None:
521  input_name = forward_op.input[input_index]
522  input_version = in_versions[input_name]
523  g = g_input[input_index]
524  if type(g) is GradientSlice:
525  # the output corresponds either to the indices or the
526  # values of the sparse gradient. In either case we
527  # create a (partial) SparseGradGenMeta. If necessary,
528  # we'll merge indices and values generators
529  # corresponding to the same gradient in step (3)
530  if g.indices == output:
531  m = SparseGradGenMeta(grad_op, i, None, 0, g)
532  else:
533  assert(g.values == output)
534  m = SparseGradGenMeta(None, 0, grad_op, i, g)
535  sparse_generators[input_name][input_version].append(m)
536  else:
537  self.gradient_generators[input_name][input_version] \
538  .append(GradGenMeta(
539  grad_op, i, g))
540 
541  # (3) merge indices and values generators for sparse gradients, and
542  # add them to gradient_generators
543  self.AppendSparseGenerators(sparse_generators)

545  # (4) for ops (e.g., Add, Sum, Sub) which have gradient outputs directly
546  # passed from inputs (not computed from gradient ops), we create an
547  # GradGenMeta with None grad_op and idx so that the gradient_generators
548  # knows where the gradients are coming from. This is needed for creating
549  # Sum op to accumulate the gradients from multiple parents.

551  input_name = forward_op.input[input_index]

556  if str(g.indices) not in locally_generated_blobs and \
557  str(g.values) not in locally_generated_blobs:
558  self.gradient_generators[input_name][input_version].append(
559  SparseGradGenMeta(None, 0, None, 0, g))
560  else:
561  if str(g) not in locally_generated_blobs:
562  self.gradient_generators[input_name][input_version].append(
563  GradGenMeta(None, 0, g))

566  # gradient frontier to reflect the input versions that the gradients
567  # correspond to.
568  for i, g in enumerate(g_input):
569  if g is not None:
570  input_name = forward_op.input[i]
571  input_version = in_versions[input_name]
572  self.gradient_frontier[input_name] = input_version

654  indices_concat_input.append(g.indices)
655  if op_v:

657  values_concat_input.append(out)
658  else:
659  self._CheckSumOpsConflict(out_base_name, g.values)
660  values_concat_input.append(g.values)
661 
662  indices_concat_output = out_base_name + '_indices_concat'
663  indices_concat_split = out_base_name + '_indices_concat_split'
664  values_concat_output = out_base_name + '_values_concat'
665  values_concat_split = out_base_name + '_values_concat_split'

667  # indices (resp. values) tensors together. We don't do any deduplication
668  # of indices at this point. This will be done as needed before the
669  # optimizer is called

679  "Concat",

681  map(BlobReference, [values_concat_output, values_concat_split]),
682  axis=0
683  ),

686  indices=indices_concat_output,
687  values=values_concat_output,

704  def _VerifyGradientGenerators(self, generator):
705  # (1) check if all gradients are of the same type. Aggregating a mix of
706  # sparse and dense gradients is not supported yet
707  if len({type(g) for g in generator}) > 1:
708  raise RuntimeError(
709  'Automatic aggregation of a mix of sparse and dense gradients '
710  'is not supported yet')
711

713  # produced the gradient, then no additional sum needs to be carried
714  # out.
715  if len(generator) < 2:
716  return False

719  all_device_options = []
720  for g in generator:
721  if type(g) is GradGenMeta:
722  if g.grad_op:

744  def DoGradientAccumulation(self, fwd_op_idx):
745  """For each input name in the forward op, check if we will need to
746  add gradient accumulation. If so, do gradient accumulation and return
747  the list of gradient operators.
748 
749  The criteria for doing gradient accumulation is:
750  (1) the specific input version has been used by multiple operators.
751  (2) the current fwd_op_idx is the first to use that input, i.e. in the
752  backward pass, is the last to optionally generate the gradient for
753  the op.
754  (3) For the operators that used the input, their gradient operators
755  have generated more than 1 gradient.
756 
757  When accumulating operators, our current solution is to rename all the
758  created gradients with an internal intermediate name, and then add a
759  Sum() operator that adds up all the gradients. This may use more memory
760  due to intermediate storage, but is usually the fastest approach as one
761  can do one single sum for multiple intermediate gradients.
762  """
763  forward_op, in_versions, out_versions = self.ssa[fwd_op_idx]
764  additional_sum_ops = []
765  grad_map = {}
766  for _i, input_name in enumerate(set(forward_op.input)):
767  input_version = in_versions[input_name]
768  input_usage = self.input_usages[input_name][input_version]
769  if (len(input_usage) <= 1 or fwd_op_idx != input_usage[0]):
770  # We do not need to do gradient accumulation yet.
771  continue
772  generator = self.gradient_generators[input_name][input_version]
773  try:
774  if not self._VerifyGradientGenerators(generator):
775  continue
776  except RuntimeError as err:
777  raise RuntimeError(
778  "Gradients for param ''{}'' failed to verify: {}".format(
779  input_name,
780  err
781  )
782  )
783 
784  # Finally, let's create the sum operator.
785  sum_ops, g = self._MakeSumOps(input_name, input_version)
786  additional_sum_ops.extend(sum_ops)
787  grad_map[input_name] = g
788  return additional_sum_ops, grad_map

808  def _GenerateGradientsForForwardOp(
809  self, forward_op_idx, input_to_grad):
810  new_input_to_grad = {}
811  gradient_ops = []
812  forward_op, in_versions, out_versions = self.ssa[forward_op_idx]
813  g_output = list(

823  for name, grad in zip(forward_op.input, g_input):

826  # we update the blob version when blob is output of an

835  def GetBackwardPass(self, ys):
836  """Gets the backward pass that computes the derivatives of given blobs.
837 
838  Inputs:
839  ys: a list or a dictionary specifying what blobs we want to compute
840  derivatives of. If the input is a list, we will automatically
841  generate their gradients with all-one values; if the input is a
842  dictionary, for any dictionary entries that are not None, we will
843  take the corresponding blobs as their gradients; for all those
844  that are None, we will auto-fill them with 1.
845  """
846  if isinstance(ys, list):
847  ys = dict((y, None) for y in ys)
848  elif not isinstance(ys, dict):
849  raise TypeError("ys should either be a list or a dict.")
850 
851  # Set the gradient frontier with the initialized external
852  # gradients.
853  for y, _ in ys.items():
854  self.gradient_frontier[y] = self.frontier[y]
855 
856  all_input_to_grad, all_gradient_ops = self._GetInitGradients(ys)
857 
858  # (2) Now, after having the virtual play above, we now play the ops
859  # backwards, creating the gradients along the path. Note that although
860  # we are playing it backwards, we cannot refer to variables that are
861  # at a version older than current_versions because it is already been
862  # overwritten.
863  for forward_op_idx in reversed(range(len(self.ssa))):
864  input_to_grad, gradient_ops = self._GenerateGradientsForForwardOp(
865  forward_op_idx, all_input_to_grad)
866  all_input_to_grad.update(input_to_grad)
867  all_gradient_ops += gradient_ops
868 
869  # If there are multiple use blobs, do gradient accumulation.
870  additional_sum_ops, grad_map = self.DoGradientAccumulation(
871  forward_op_idx)
872  # This line is so that if in an accumulation some of the operators
873  # have not produced gradients, they still do not overwrite the
874  # general all_input_to_grad map.
875  all_input_to_grad.update(grad_map)
876  all_gradient_ops += additional_sum_ops
877 
878  # (3) Post-processing.
879  # After we have done computation for each op, we now have the gradient
880  # operators ready. For the output map, we will convert everything to
881  # BlobReferences for easier handling in python.
882  all_input_to_grad_out = {}
883  for key, val in all_input_to_grad.items():
884  if val is not None:
885  all_input_to_grad_out[BlobReference(key)] = (
886  BlobReference(val) if isinstance(val, basestring) else
887  GradientSlice(BlobReference(val[0]), BlobReference(val[1])))
888  return all_gradient_ops, all_input_to_grad_out
