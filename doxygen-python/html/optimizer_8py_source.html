24  @staticmethod
25  def build_lr(net, param_init_net, base_learning_rate,
26  learning_rate_blob="lr", policy="fixed",
27  iter_val=0, **kwargs):
28  if not param_init_net.BlobIsDefined(_OPTIMIZER_ITERATION_NAME):
29  # Add training operators.
30  with core.DeviceScope(core.DeviceOption(caffe2_pb2.CPU)):
31  iteration = param_init_net.ConstantFill(
32  [], _OPTIMIZER_ITERATION_NAME, shape=[1],
33  value=iter_val,
34  dtype=core.DataType.INT64)
35  iter_mutex = param_init_net.CreateMutex([], ["iteration_mutex"])
36  net.AtomicIter([iter_mutex, iteration], [iteration])
37  else:
38  iteration = param_init_net.GetBlobRef(_OPTIMIZER_ITERATION_NAME)
39 
40  # There is one interesting thing here: since we are minimizing, we are
41  # doing "descent" so the learning rate is set to be negative.
42  lr = net.LearningRate(
43  [iteration],
44  learning_rate_blob,
45  base_lr=-base_learning_rate,
46  policy=policy,
47  **kwargs
48  )
49  return lr, iteration

60  def get_auxiliary_parameters(self):
61  """Returns a list of auxiliary parameters.
62 
63  Returns:
64  aux_params: A namedtuple, AuxParams.
65 
66  aux_params.local stores a list of blobs. Each blob is a local
67  auxiliary parameter. A local auxiliary parameter is a parameter in
68  parallel to a learning rate parameter. Take adagrad as an example,
69  the local auxiliary parameter is the squared sum parameter, because
70  every learning rate has a squared sum associated with it.
71 
72  aux_params.shared also stores a list of blobs. Each blob is a shared
73  auxiliary parameter. A shared auxiliary parameter is a parameter
74  that is shared across all the learning rate parameters. Take adam as
75  an example, the iteration parameter is a shared parameter, because
76  all the learning rates share the same iteration parameter.
77  """
78  return self._aux_params

93 class SgdOptimizer(Optimizer):
94  def __init__(self, base_learning_rate=0.01, policy='fixed',
95  momentum=0.0, **kwargs):
96  super(SgdOptimizer, self).__init__()
97  self.base_learning_rate = base_learning_rate
98  self.policy = policy
99  self.momentum = momentum
100  self.init_kwargs = kwargs
101 
102  def __call__(self, net, param_init_net, param, grad):
103  if self.base_learning_rate <= 0:
104  return
105 
106  lr, _ = self.build_lr(
107  net, param_init_net,
108  base_learning_rate=self.base_learning_rate,
109  learning_rate_blob=str(param) + "_lr",
110  policy=self.policy,
111  **(self.init_kwargs)
112  )
113 
114  ONE = param_init_net.ConstantFill([], "ONE", shape=[1], value=1.0)
115  self._aux_params.shared.append(ONE)
116 
117  if self.momentum > 0:
118  momentum_data = param_init_net.ConstantFill(
119  param, str(param) + "_momentum", value=0.)
120  self._aux_params.local.append(momentum_data)
121 
122  if isinstance(grad, core.GradientSlice):
123  assert self.momentum == 0., "Doesn't support momentum for sparse"
124  net.ScatterWeightedSum(
125  [param, ONE, grad.indices, grad.values, lr],
126  param
127  )
128  else:
129  if self.momentum > 0.:
130  net.MomentumSGD(
131  [grad, momentum_data, lr], [grad, momentum_data],
132  momentum=self.momentum,
133  nesterov=1)
134  coeff = ONE
135  else:
136  coeff = lr
137 
138  net.WeightedSum(
139  [param, ONE, grad, coeff],
140  param
141  )
