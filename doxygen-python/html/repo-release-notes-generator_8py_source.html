78 def generateTable(url, params, limit, c):
79  """
80  Creates a markdown table from Github issues
81  """

83  def bodyBreaker(body, width):
84  parts = body.split()
85  # Break the body into parts but only keep the parts
86  body = ""
87  for word in parts:
88  if len(word) > width:
89  # If word is longer than 64 chars then trim it
90  word = (word[:width] + '..')
91  # Resurrection of the body maybe missing a little bit
92  body += word + ' '
93  return body

95  # Load and prep the json data from the API
96  data = requests.get(url=url, params=params)
97  binary = data.content
98  output = json.loads(binary)
99  # Start the counter for the number of issues loaded
100  total = 0
101  # Create a list of user, user_url entries
102  contributors = {}
103 
104  label = params.get("labels").capitalize()
105  if label == "Bug":
106  label = "Bug Fix"
107  print("\n## {} Updates".format(label))
108  intro = "\n| Title | Summary | Contributor |\n|---|---|---|"
109  print(intro)
110 
111  for result in output:
112  # Filter PRs optional --> if 'pull_request' in result:
113  # Non-UTF-8 characters sometimes appear and cause issues
114  # Line breaks will break the markdown
115  title = result['title'].encode('utf-8').replace('\n', ' ') \
116  .replace('\r', '')
117  title = bodyBreaker(title, 25)
118  url = result['html_url']
119  # Trim the body to the max limit and remove bad stuff
120  # Backticks with long code blocks mess up formatting
121  body = result['body'][0:limit].encode('utf-8').replace('\n', ' ') \
122  .replace('\r', '').replace('`', '')
123  # If the body is too long add ... to show there is more
124  if (len(result['body']) > limit):
125  body += '...'
126  # Sometimes there are long words that force the column widths too wide
127  body = bodyBreaker(body, 50)
128  # Naughty developers that don't add any details get an n/a
129  if len(body) == 0:
130  body = 'n/a'
131  user = result['user']['login'].encode('utf-8')
132  user_url = result['user']['html_url']
133  # Add user to contributors list
134  contributors[user] = user_url
135  c[user] += 1
136  print("|[{}]({})|{}|[{}]({})|".format(title, url, body, user, user_url))
137  total += 1
138 
139  # Check output totals
140  print("Total of {} out of {}\n".format(total, len(output)))
141  return contributors, c
