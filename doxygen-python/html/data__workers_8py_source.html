9 '''
10 This module provides a python-land multithreaded data input mechanism
11 for Caffe2 nets.
12 
13 Basic usage is as follows:
14  coordinator = data_workers.init_data_input_workers(
15  net,
16  ["data", "label"],
17  my_fetch_fun,
18  batch_size=32,
19  input_source_name="train",
20  dont_rebatch=False
21  )
22  ...
23  coordinator.start()
24 
25 First argument is the Caffe2 net (or model helper), and second argument
26 is list of input blobs that are to be fed.
27 
28 Argument 'input_source_name' is used to distinguish different sources of data,
29 such as train or test data. This is to ensure the data does not get mixed up,
30 although two nets would share blobs.
31 
32 To do the actual data loading, one defines a "fetcher function"
33 that has call signature
34  my_fetch_fun(worker_id, batch_size)
35 
36 Optionally, one can define a "init function" that is called once before
37 threads start, and has call signature:
38  my_init_fun(data_coordinator, global_coordinator)
39 
40 If dont_rebatch is set to True, the data input is not batched into equal sized
41 chunks but data directly provided by fetchers is used.
42 
43 'batch_columns' can be used to specify which dimension is the batch dimension,
44 for each of the inputs. Default is 0 for all iputs.
45 
46 This function returns a list of numpy arrays corresponding to the different
47 input blobs. In the example above, it would return two arrays, one for the
48 data blob and another for the labels. These arrays can have arbitrary number
49 of elements (i.e they do not need to match the batch size). The batch size
50 is provided for the function as a hint only.
51 
52 For example, fetcher function could download images from a remote service or
53 load random images from a directory on a file system.
54 
55 For a dummy example, see the data_workers_test unit test.
56 
57 Note that for data_parallel_models, init_data_input_workers will be called
58 for each GPU. Note that the 'coordinator' returned by the function is same
59 each time.
60 '''

91  external_loggers=None,
92  dont_rebatch=False,
93  batch_columns=None,
94 ):
95  global global_coordinator
96  device_option = scope.CurrentDeviceScope()

98  device_option = caffe2_pb2.DeviceOption(device_type=caffe2_pb2.CPU)

100  # Create coordinator object
101  coordinator = DataInputCoordinator(
102  net,

104  batch_size,

106  scope.CurrentNameScope(),
107  input_source_name,
108  global_coordinator.get_queue(input_source_name, max_buffered_batches),

110  external_loggers=external_loggers,
111  dont_rebatch=dont_rebatch,
112  batch_columns=batch_columns,
113  )
114 
115  # Launch fetch worker threads
116  worker_ids = [
117  global_coordinator.get_new_worker_id()
118  for i in range(num_worker_threads)
119  ]
120  workers = [
121  threading.Thread(
122  target=fetcher,
123  name="data_workers fetcher id {}".format(worker_id),
124  args=[coordinator, worker_id, fetch_fun, batch_size, input_blob_names],
125  ) for worker_id in worker_ids
126  ]

128  workers.append(threading.Thread(
129  target=enqueuer,
130  name="Enqueuer {} {}".format(input_source_name, scope.CurrentNameScope()),

207  print("Wait for workers to die: {}".format(self._input_source_name))
208  for w in self._workers:
209  if w != threading.current_thread():

220  def _get(self):
221  while self.is_active():
222  try:
223  return self._internal_queue.get(block=True, timeout=0.5)
224  except Queue.Empty:
225  continue
226  return None

228  def put(self, chunk):
229  if len(chunk) == 0:
230  print("Worker provided zero length input")
231  return
232  while self.is_active():
233  try:
234  qsize = self._internal_queue.qsize()
235  if qsize < 2 and (time.time() - self._last_warning) > LOG_INT_SECS:
236  print("Warning, data loading lagging behind: " +
237  "name={}".format(qsize, self._input_source_name))
238  self._last_warning = time.time()
239  self._counter += 1
240  self._internal_queue.put(chunk, block=True, timeout=0.5)
241  self._log_inputs_per_interval(chunk[0].shape[0])
242  return
243  except Queue.Full:
244  log.debug("Queue full: stalling fetchers...")
245  continue

255  def _enqueue_batch(self):
256  '''
257  This pulls data from the python-side queue and collects them
258  into batch-sized pieces, unless dont_rebatch is set to true.
259  '''
260  if self._dont_rebatch:
261  self._enqueue_batch_direct()
262  return
263 
264  cur_batch = [np.array([]) for d in self._input_blob_names]
265  first_batch_col = self._batch_columns[0]
266 
267  # Collect data until we have a full batch size
268  while (
269  cur_batch[0].shape[0] == 0 or
270  cur_batch[0].shape[first_batch_col] < self._batch_size
271  ) and self.is_active():
272  chunk = self._get()
273  if chunk is None:
274  continue
275 
276  for j, chunk_elem in enumerate(chunk):
277  if cur_batch[j].shape[0] == 0:
278  cur_batch[j] = chunk_elem.copy()
279  else:
280  cur_batch[j] = np.append(
281  cur_batch[j], chunk_elem, axis=self._batch_columns[j]
282  )
283 
284  start_time = time.time()
285  try:
286  # Return data over the batch size back to queue
287  if cur_batch[0].shape[0] > 0 and cur_batch[0].shape[
288  first_batch_col
289  ] > self._batch_size:
290  leftover = []
291  trimmed_batch = []
292  for j, b in enumerate(cur_batch):
293  [c, l] = np.split(
294  b, [self._batch_size], axis=self._batch_columns[j]
295  )
296  leftover.append(l)
297  trimmed_batch.append(c)
298  cur_batch = trimmed_batch
299  try:
300  self._internal_queue.put(leftover, block=False)
301  except Queue.Full:
302  pass
303 
304  assert cur_batch[0].shape[first_batch_col] == self._batch_size
305 
306  if self.is_active():
307  for b, q, c in zip(
308  self._input_blob_names, self._queues, cur_batch
309  ):
310  self._enqueue(b, q, c)
311  finally:
312  self.put_metric('enqueue_time', time.time() - start_time)

314  def _enqueue(self, blob_name, queue, data_arr):
315  '''
316  Enqueue the correctly sized batch arrays to Caffe2's queue.
317  '''
318  scratch_name = self._namescope + blob_name + \
319  "_scratch_" + self._input_source_name
320  blob = core.BlobReference(scratch_name)
321  status = core.BlobReference(scratch_name + "_status")
322  workspace.FeedBlob(
323  blob,
324  data_arr,
325  device_option=self._device_option
326  )
327  self._scratch_blobs.add(blob)
328  self._scratch_blobs.add(status)
329 
330  op = core.CreateOperator(
331  "SafeEnqueueBlobs",
332  [queue, blob],
333  [blob, status],
334  device_option=self._device_option
335  )
336  workspace.RunOperatorOnce(op)

474 def fetcher(coordinator, worker_id, fetch_fun, batch_size, input_blob_names):
475  while coordinator.is_active():
476  start_time = time.time()
477  try:
478  input_data = fetch_fun(worker_id, batch_size)
479  if input_data is None:
480  print("Fetcher function returned None")
481  continue
482 
483  assert len(input_data) == len(input_blob_names), \
484  "Expecting data blob for each input"
485  for d in input_data:
486  assert isinstance(d, np.ndarray), \
487  "Fetcher function must return a numpy array"
488  if not coordinator._dont_rebatch:
489  j = 1
490  for d in input_data[1:]:
491  assert d.shape[coordinator._batch_columns[j]] == input_data[0].shape[coordinator._batch_columns[0]], \
492  "Each returned input must have equal number of samples"
493  j += 1
494 
495  coordinator.put(input_data)
496  except Exception as e:
497  print(e)
498  logging.exception("Exception in fetcher", e)
499  coordinator._stop("Exception in fetcher {}: {}".format(
500  worker_id, e
501  ))
502  finally:
503  coordinator.put_metric('fetcher_time', time.time() - start_time)
