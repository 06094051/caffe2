35  Function to create a model that can run on many GPUs.
36  model_helper_obj: an object of ModelHelper, such as CNNModelHelper

38  Function that adds the input operators
39  Note: Remember to instantiate reader outside of this

43  Function to add the operators to the model.

51  gradient update, such as updating the weights and
52  weight decaying.

55  rendezvous: used for rendezvous in distributed computation, if None
56  then only one node is used. To create rendezvous,
57  use <TBD>.

59  optimize_gradient_memory: whether to apply 'memonger' to share blobs
60  in gradient computation to reduce memory footprint

209 def Parallelize_GPU_BMUF(
210  model_helper_obj,
211  input_builder_fun,
212  forward_pass_builder_fun,
213  param_update_builder_fun,
214  block_learning_rate=1.0,
215  block_momentum=None,
216  devices=range(0, workspace.NumCudaDevices()),
217  net_type='dag',
218  master_gpu=None,
219 ):
220  '''
221  Function to create model that run on many GPUs and creates a net for
222  parameter_updates that can be run independently for number of iterations
223  then followed by another net that runs once to compute the final parameter
224  updates according to block wise model update filtering rule described
225  in : Scalable Training of Deep Learning Machines by Incremental Block
226  Training with Intra-block Parallel Optimization and Blockwise Model-Update
227  Filtering (ICASSP 2016).
228  '''
229  assert isinstance(model_helper_obj, model_helper.ModelHelper)
230 
231  if master_gpu is None:
232  master_gpu = devices[0]
233 
234  model_helper_obj._devices = devices
235  master_gpu_opt = core.DeviceOption(caffe2_pb2.CUDA, master_gpu)
236 
237  num_workers = len(devices)
238  loss_scale = 1.0 / num_workers
239  if block_momentum is None:
240  block_momentum = 1.0 - 1.0 / num_workers
241 
242  model_helper_obj.net.Proto().num_workers = num_workers
243  model_helper_obj.net.Proto().type = net_type
244 
245  # A net for initializing global model parameters. Its called once in the
246  # same step as net parameters initialization.
247  model_helper_obj._global_model_init_net = core.Net('global_model_init')
248  model_helper_obj._global_model_init_net.Proto().type = net_type
249  model_helper_obj._global_model_init_net.Proto().num_workers = num_workers
250 
251  # A net for computing final parameter updates. Its will run once after
252  # running net (local models updates) for `num_local_iterations` times.
253  model_helper_obj._global_model_param_updates_net = core.Net('global_model')
254  model_helper_obj._global_model_param_updates_net.Proto().type = net_type
255  model_helper_obj._global_model_param_updates_net.Proto().num_workers = \
256  num_workers
257 
258  def _v(param):
259  return "{}_v".format(param)
260 
261  def _g(param):
262  return "{}_g".format(param)
263 
264  # Keep track of params that were in the model before: they are not
265  # data parallel, so we need to handle them separately
266  non_datapar_params = copy.copy(model_helper_obj.params)
267  model_helper_obj._losses_by_gpu = {}
268 
269  def _InitializeModels(gpu_id):
270  input_builder_fun(model_helper_obj)
271  loss = forward_pass_builder_fun(model_helper_obj, loss_scale)
272  model_helper_obj._losses_by_gpu[gpu_id] = loss
273  _ForEachGPU(devices, _InitializeModels, scoped=True)
274 
275  model_helper_obj._device_grouped_blobs =\
276  _GroupByDevice(devices, model_helper_obj.params, non_datapar_params)
277 
278  _AddGradientOperators(
279  devices, model_helper_obj, model_helper_obj._losses_by_gpu
280  )
281 
282  _InferBlobDevice(model_helper_obj)
283 
284  def _InitializeParamUpdate(gpu_id):
285  param_update_builder_fun(model_helper_obj)
286  _ForEachGPU(devices, _InitializeParamUpdate, scoped=True)
287 
288  # (Step-0) Initialize momentum parameters on master GPU.
289  for param_name in model_helper_obj._device_grouped_blobs.keys():
290  param = model_helper_obj._device_grouped_blobs[param_name][master_gpu]
291  with core.DeviceScope(master_gpu_opt):
292  model_helper_obj._global_model_init_net.ConstantFill(
293  param, _v(param), value=0.0
294  )
295  model_helper_obj._global_model_init_net.Copy(param, _g(param))
296 
297  # (Step-1) Update models for num_local_iterations.
298 
299  # (Step-2) Comute post-local-updates average of the params.
300  # Sum model params across GPUs and store resutls in param_avg blob.
301  for param_name in model_helper_obj._device_grouped_blobs.keys():
302  with core.DeviceScope(master_gpu_opt):
303  _AllReduce(
304  devices, model_helper_obj,
305  model_helper_obj._global_model_param_updates_net,
306  param_name
307  )
308 
309  # (Step-3) Update momentum params :
310  # param_v = block_momentum * param_v
311  # + block_learning_Rate * (param_avg - param)
312  # param = param + param_v
313  for param_name in model_helper_obj._device_grouped_blobs.keys():
314  param = model_helper_obj._device_grouped_blobs[param_name][master_gpu]
315  with core.DeviceScope(master_gpu_opt):
316  # TODO(ataei) : Stop building the graph here to get model average ?
317  model_helper_obj._global_model_param_updates_net.Scale(
318  param, param, scale=1.0 / num_workers
319  )
320  model_helper_obj._global_model_param_updates_net.Sub(
321  [param, _g(param)], param
322  )
323  model_helper_obj._global_model_param_updates_net.Scale(
324  param, param, scale=block_learning_rate
325  )
326  model_helper_obj._global_model_param_updates_net.Scale(
327  _v(param), _v(param), scale=block_momentum
328  )
329  model_helper_obj._global_model_param_updates_net.Add(
330  [_v(param), param], _v(param)
331  )
332  model_helper_obj._global_model_param_updates_net.Add(
333  [_g(param), _v(param)], _g(param)
334  )
335  model_helper_obj._global_model_param_updates_net.Copy(
336  _g(param), param
337  )
338  _Broadcast(
339  devices, model_helper_obj,
340  model_helper_obj._global_model_param_updates_net,
341  param_name
342  )

511  def sum2(d1i, d2i):
512  d1 = model._devices[d1i]
513  d2 = model._devices[d2i]
514  device_opt = core.DeviceOption(caffe2_pb2.CUDA, d1)
515  with core.DeviceScope(device_opt):

525  sum2(j * 4, j * 4 + 2)

529  sum2(0, 1)

531  sum2(0, 2)

619  # We need to specify a partial order using control_input to ensure
620  # progress (all machines need to do same allreduce in parallel)
621  num_controls = min(4, num_workers - 1)
622  cyclical_controls = []
623 
624  # Since num_controls determines the partial ordering of
625  # allreduces, there is no need for more common world instances
626  # than there are parallel allreduce operations.
627  num_comm_worlds = num_controls
628  cyclical_comm_worlds = []
629 
630  counter = 0
631  nccl_control_blob = None
632 
633  # Note: sorted order to ensure each host puts the operators in
634  # same order.
635  for grad_name in reverse_ordered_grads:
636  master_grad = model._device_grouped_blobs[grad_name][devices[0]]
637  grads_group = model._device_grouped_blobs[grad_name].values()
638 
639  assert master_grad in grads_group
640 
641  # Remark: NCCLReduce does not support in-place modifications
642  # so we need a temporary gradient blob
643  reduced_grad = str(master_grad) + "_red"
644 
645  control_input = None if len(cyclical_controls) < num_controls \
646  else cyclical_controls[counter % num_controls]
647  comm_world = None if len(cyclical_comm_worlds) < num_comm_worlds \
648  else cyclical_comm_worlds[counter % num_comm_worlds]
649 
650  def allreduce(comm_world, grads):
651  with core.DeviceScope(reducing_device_opt):
652  if comm_world is None:
653  comm_number = len(cyclical_comm_worlds)
654  comm_world = model.param_init_net.CreateCommonWorld(
655  rendezvous['kv_handler'],
656  "allreduce_{}_cw".format(comm_number),
657  name="allreduce_{}_cw_op".format(comm_number),
658  size=rendezvous['num_shards'],
659  rank=rendezvous['shard_id'],
660  engine=rendezvous['engine'],
661  status_blob="create_cw_{}_status".format(comm_number),
662  )
663  model.net.Allreduce(
664  inputs=[comm_world] + grads,
665  outputs=grads,
666  name=grad_name,
667  engine=all_reduce_engine,
668  control_input=control_input,
669  status_blob="allreduce_{}_status".format(grad_name),
670  )
671  return comm_world
672 
673  if rendezvous['engine'] == 'GLOO':
674  # With Gloo cross GPU and cross machine allreduce
675  # can be executed in a single operation
676  comm_world = allreduce(comm_world, grads_group)
677  control_output = grads_group[0]
678  else:
679  # Step 1: sum gradients from local GPUs to master GPU
680  with core.DeviceScope(master_device_opt):
681  model.ConstantFill(master_grad, reduced_grad, value=0.0)
682 
683  # Temp fix since NCCLReduce does not work
684  model.net.NCCLAllreduce(
685  grads_group,
686  grads_group,
687  control_input=nccl_control_blob,
688  )
689  nccl_control_blob = grads_group[0]
690  model.net.Copy(master_grad, reduced_grad)
691 
692  # Step 2: allreduce between all hosts, between master GPUs
693  comm_world = allreduce(comm_world, [reduced_grad])
694  control_output = reduced_grad
695 
696  with core.DeviceScope(master_device_opt):
697  model.net.Copy(reduced_grad, master_grad)
698 
699  # Step 3: broadcast locally
700  _Broadcast(devices, model, model.net, grad_name)
701 
702  if len(cyclical_controls) < num_controls:
703  cyclical_controls.append(control_output)
704  else:
705  cyclical_controls[counter % num_controls] = control_output
706 
707  if len(cyclical_comm_worlds) < num_comm_worlds:
708  cyclical_comm_worlds.append(comm_world)
709  else:
710  assert cyclical_comm_worlds[counter % num_comm_worlds] == comm_world
711 
712  counter += 1

715 def _AllReduceGradientsSingleHost(devices, model, use_nccl):
716  """Performs NCCL AllReduce to distribute gradients to all the GPUs."""
717 
718  if len(devices) == 1:
719  return
720 
721  # Gradients in reverse order
722  reverse_ordered_grads = _GetReverseOrderedGrads(model)
723  assert(len(reverse_ordered_grads) > 0)
724 
725  # Now we need to Allreduce gradients on all the GPUs.
726  # Pick GPU #0 as a master GPU.
727  master_device_opt = core.DeviceOption(caffe2_pb2.CUDA, devices[0])
728  last_out = None
729 
730  for grad_name in reverse_ordered_grads:
731  # Group by grads for reduce.
732  grads_group = model._device_grouped_blobs[grad_name].values()
733  assert len(grads_group) == len(devices), \
734  "Each GPU from {}, should have a copy of {}.".format(
735  devices, grad_name)
736 
737  if _IsGPUBlob(model, grad_name):
738  with core.DeviceScope(master_device_opt):
739  if not isinstance(grads_group[0], core.GradientSlice):
740  _AllReduce(
741  devices, model, model.net, grad_name, use_nccl, last_out
742  )
743  # last_out is used to serialize the execution of nccls
744  last_out = grads_group[0]
745 
746  else:
747  # Sparse gradients: all-gather for indices and values
748  master_ns = "gpu_{}".format(devices[0])
749  grad_idx_concat, _ = model.net.Concat(
750  [g.indices for g in grads_group],
751  ["{}/{}_index_concat".format(master_ns, grad_name),
752  "{}/{}_index_splitinfo".format(master_ns, grad_name)],
753  axis=0,
754  name="note:data_parallel_model")
755  for gpu, g in model._device_grouped_blobs[grad_name].items():
756  device_opt = core.DeviceOption(caffe2_pb2.CUDA, gpu)
757  with core.DeviceScope(device_opt):
758  model.Copy(grad_idx_concat, g.indices)
759 
760  grad_val_concat, _ = model.net.Concat(
761  [g.values for g in grads_group],
762  ["{}/{}_val_concat".format(master_ns, grad_name),
763  "{}/{}_val_splitinfo".format(master_ns, grad_name)],
764  axis=0, name="note:data_parallel_model")
765  for gpu, g in model._device_grouped_blobs[grad_name].items():
766  device_opt = core.DeviceOption(caffe2_pb2.CUDA, gpu)
767  with core.DeviceScope(device_opt):
768  model.Copy(grad_val_concat, g.values)
769 
770  else:
771  assert isinstance(grads_group[0], core.GradientSlice), \
772  "Synchronizing gradient slices not supported"
773  with core.DeviceScope(core.DeviceOption(caffe2_pb2.CPU)):
774  # Poor man's allreduce
775  model.Sum(grads_group, grads_group[0])
776  _Broadcast(devices, model, grad_name)

891 def _GroupByDevice(devices, params, non_data_params):
892  '''
893  Groups blobs by device, returning a map of [blobname] = {0: BlobRef, 1: ..}.
894  Returns ordered dictionary, ensuring the original order.
895  '''
896  grouped = OrderedDict()
897  # Only consider params that were created to be "data parallel"
898  params = params[len(non_data_params):]
899  assert len(params) % len(devices) == 0,\
900  "There should be equal number of params per device"
901 
902  num_params_per_device = int(len(params) / len(devices))
903 
904  for i, p in enumerate(params):
905  assert isinstance(p, core.BlobReference) or \
906  isinstance(p, core.GradientSlice), \
907  "Param {} is not BlobReference or GradientSlice".format(p)
908 
909  name = stripParamName(p)
910  gpuid = devices[i // num_params_per_device]
911 
912  if isinstance(p, core.BlobReference):
913  assert "gpu_{}/".format(gpuid) in p.GetNameScope(),\
914  "Param {} expected to have namescope 'gpu_{}'".format(str(p), gpuid)
915  else:
916  assert "gpu_{}/".format(gpuid) in p.indices.GetNameScope(),\
917  "Indices {} expected to have namescope 'gpu_{}'".format(str(p), gpuid)
918  assert "gpu_{}/".format(gpuid) in p.values.GetNameScope(),\
919  "Values {} expected to have namescope 'gpu_{}'".format(str(p), gpuid)
920 
921  if name not in grouped:
922  grouped[name] = {}
923  grouped[name][gpuid] = p
924 
925  # Confirm consistency
926  for j, (p, ps) in enumerate(grouped.items()):
927  assert \
928  len(ps) == len(devices), \
929  "Param {} does not have value for each device (only {}: {})".format(
930  p, len(ps), ps,
931  )
932  # Ensure ordering
933  if (ps[devices[0]] != params[j]):
934  log.error("Params: {}".format(params))
935  log.error("Grouped: {}".format(grouped.keys()))
936  assert ps[devices[0]] == params[j], \
937  "Incorrect ordering: {}".format(ps)
938 
939  return grouped
